const Counter = create("counter", {
	state: () => ({ count: 0 }),

	render: (state, props, children) => `
    <div class="counter">
      <div>Count: ${state.count.get()}</div>
      <button>+</button>
      ${children || ""}
    </div>
  `,

	init: (element, state) => {
		element.querySelector("button")?.addEventListener("click", () => {
			state.count.update((n) => n + 1);
		});
	},
});
